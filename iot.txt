{

}
void loop()
{

{

}
else
{

}
}




import RPi.GPIO as GPIO
import time
import Adafruit_DHT

GPIO.setmode(GPIO.BCM)
GPIO.setup(4, GPIO.IN)

while True:
    hum, temp = Adafruit_DHT.read_retry(Adafruit_DHT.DHT11, 4)
    print("Temp:", temp)
    print("Humidity:", hum)

    if temp is not None and temp > 30:
        print("⚠ Alert: Temperature above 30°C!")

    if hum is not None and hum > 60:
        print("⚠ Alert: Humidity above 60%!")

    time.sleep(2)
